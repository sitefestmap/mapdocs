```php
$array = [1, 2, 3];
// List syntax

// shorthand
[$a, $b, $c] = $array

// skip elements
[, , $c] = array

// keys
['c' => $c, 'a' => $a] = array 
```

```php
<?php
    $name = array(
        firstname => John,
        lastname => Smith
    );
?>

// Short Array Syntax
<?php
    $name = [
        firstname => John,
        lastname => Smith
    ];
?>
```
